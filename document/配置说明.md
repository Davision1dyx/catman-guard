# 智能运维Agent系统配置说明

## 1. 数据库配置

### 1.1 PostgreSQL + PGVector

```yaml
spring:
  datasource:
    url: jdbc:postgresql://localhost:5432/ops_agent
    username: postgres
    password: postgres
    driver-class-name: org.postgresql.Driver
  
  jpa:
    database-platform: org.hibernate.dialect.PostgreSQLDialect
    show-sql: false
    hibernate:
      ddl-auto: update
  
  # PGVector配置
  ai:
    vectorstore:
      pgvector:
        dimensions: 1536                    # 向量维度(根据embedding模型调整)
        distance-type: cosine_distance      # 相似度计算方式
        index-type: hnsw                    # 索引类型(hnsw高性能)
        max-document-batch-size: 10000      # 批量插入大小
        initialize-schema: true              # 自动创建表
        table-name: vector_store            # 表名
```


### 2.3 缓存配置类

```java
@Configuration
@EnableCaching
public class CacheConfig {
    
    /**
     * 会话缓存
     */
    @Bean
    public Cache<String, SessionData> sessionCache() {
        return Caffeine.newBuilder()
            .maximumSize(10000)
            .expireAfterWrite(7, TimeUnit.DAYS)
            .recordStats()
            .build();
    }
    
    /**
     * 用户信息缓存
     */
    @Bean
    public Cache<Long, UserInfo> userInfoCache() {
        return Caffeine.newBuilder()
            .maximumSize(5000)
            .expireAfterWrite(24, TimeUnit.HOURS)
            .recordStats()
            .build();
    }
    
    /**
     * 用户权限缓存
     */
    @Bean
    public Cache<Long, Set<String>> userPermissionsCache() {
        return Caffeine.newBuilder()
            .maximumSize(5000)
            .expireAfterWrite(24, TimeUnit.HOURS)
            .recordStats()
            .build();
    }
    
    /**
     * 文件处理状态缓存
     */
    @Bean
    public Cache<String, FileStatus> fileStatusCache() {
        return Caffeine.newBuilder()
            .maximumSize(10000)
            .expireAfterWrite(1, TimeUnit.HOURS)
            .recordStats()
            .build();
    }
    
    /**
     * 热门问题缓存
     */
    @Bean
    public Cache<String, Integer> hotQuestionsCache() {
        return Caffeine.newBuilder()
            .maximumSize(100)
            .expireAfterWrite(1, TimeUnit.DAYS)
            .recordStats()
            .build();
    }
}
```

### 2.4 缓存使用示例

```java
@Service
public class CacheService {
    
    @Autowired
    private Cache<String, SessionData> sessionCache;
    
    @Autowired
    private Cache<Long, UserInfo> userInfoCache;
    
    /**
     * 获取会话数据
     */
    public SessionData getSession(String sessionId) {
        return sessionCache.get(sessionId, key -> {
            // 缓存未命中,从数据库加载
            return loadSessionFromDB(key);
        });
    }
    
    /**
     * 保存会话数据
     */
    public void saveSession(String sessionId, SessionData data) {
        sessionCache.put(sessionId, data);
        // 同时持久化到数据库
        saveSessionToDB(data);
    }
    
    /**
     * 清除会话缓存
     */
    public void clearSession(String sessionId) {
        sessionCache.invalidate(sessionId);
    }
    
    /**
     * 获取缓存统计信息
     */
    public Map<String, Object> getCacheStats() {
        CacheStats stats = sessionCache.stats();
        Map<String, Object> result = new HashMap<>();
        result.put("hitCount", stats.hitCount());
        result.put("missCount", stats.missCount());
        result.put("hitRate", stats.hitRate());
        result.put("evictionCount", stats.evictionCount());
        return result;
    }
}
```

### 3.3 文件存储服务

```java
@Service
public class FileStorageService {
    
    @Value("${file.storage.local.base-path}")
    private String basePath;
    
    /**
     * 保存文件
     */
    public String saveFile(MultipartFile file, String category) throws IOException {
        // 生成文件ID和存储路径
        String fileId = UUID.randomUUID().toString();
        String fileName = file.getOriginalFilename();
        String filePath = buildFilePath(category, fileId, fileName);
        
        // 创建目录
        File destFile = new File(basePath + filePath);
        destFile.getParentFile().mkdirs();
        
        // 保存文件
        file.transferTo(destFile);
        
        return filePath;
    }
    
    /**
     * 构建文件路径
     * 格式: /uploads/2026/04/27/{category}/{fileId}_{fileName}
     */
    private String buildFilePath(String category, String fileId, String fileName) {
        LocalDate today = LocalDate.now();
        String datePath = today.format(DateTimeFormatter.ofPattern("yyyy/MM/dd"));
        return String.format("uploads/%s/%s/%s_%s", datePath, category, fileId, fileName);
    }
    
    /**
     * 读取文件
     */
    public InputStream readFile(String filePath) throws FileNotFoundException {
        File file = new File(basePath + filePath);
        return new FileInputStream(file);
    }
    
    /**
     * 删除文件
     */
    public boolean deleteFile(String filePath) {
        File file = new File(basePath + filePath);
        return file.delete();
    }
    
    /**
     * 检查文件是否存在
     */
    public boolean exists(String filePath) {
        File file = new File(basePath + filePath);
        return file.exists();
    }
    
    /**
     * 获取文件大小
     */
    public long getFileSize(String filePath) {
        File file = new File(basePath + filePath);
        return file.length();
    }
}
```

```bash

sudo yum install postgresql-pgvector

# 初始化数据库
sudo postgresql-setup initdb

# 启动PostgreSQL
sudo systemctl start postgresql
sudo systemctl enable postgresql

# 创建数据库和用户
sudo -u postgres psql

CREATE USER ops_user WITH PASSWORD 'your_password';
GRANT ALL PRIVILEGES ON DATABASE ops_agent TO ops_user;

# 连接数据库并安装扩展
\c ops_agent
CREATE EXTENSION IF NOT EXISTS vector;
```

## 8. 性能调优建议

### 8.1 Caffeine缓存优化

- 根据实际访问模式调整TTL和最大容量
- 监控缓存命中率,保持在80%以上
- 使用异步刷新避免缓存击穿
- 合理设置权重,避免内存溢出

### 8.2 文件存储优化

- 使用SSD磁盘提升IO性能
- 定期清理临时文件和过期文件
- 使用符号链接支持多磁盘存储
- 监控磁盘空间使用情况

### 8.3 数据库优化

- 为高频查询字段创建索引
- 定期执行VACUUM和ANALYZE
- 配置合适的连接池大小
- 监控慢查询并优化

## 9. 故障排查

### 9.1 缓存问题

**问题**: 缓存命中率低
**解决**:
- 检查缓存配置是否合理
- 查看缓存统计信息
- 调整TTL和最大容量

### 9.2 文件存储问题

**问题**: 文件上传失败
**解决**:
- 检查目录权限
- 检查磁盘空间
- 查看应用日志

### 9.3 数据库问题

**问题**: 向量检索慢
**解决**:
- 检查PGVector索引是否创建
- 优化查询参数(topK、阈值)
- 增加数据库内存配置
